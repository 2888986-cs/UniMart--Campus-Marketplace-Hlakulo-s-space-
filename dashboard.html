function renderMonthlyChart(monthly) { const canvas = document.getElementById('monthly-chart'); const ctx = canvas.getContext('2d'); const max = Math.max(...monthly.map(m => m.value), 1); const W = canvas.offsetWidth; const H = 160; canvas.width = W; canvas.height = H; const barW = Math.floor(W / monthly.length * 0.5); const gap = Math.floor(W / monthly.length); ctx.clearRect(0, 0, W, H); monthly.forEach((m, i) => { const x = i * gap + gap * 0.25; const barH = Math.max((m.value / max) * (H - 30), 2); ctx.fillStyle = '#3a6645'; ctx.beginPath(); ctx.roundRect(x, H - barH - 20, barW, barH, [4, 4, 0, 0]); ctx.fill(); ctx.fillStyle = '#6b6b5a'; ctx.font = '10px Inter, sans-serif'; ctx.textAlign = 'center'; ctx.fillText(m.label, x + barW / 2, H - 4); }); } function renderCategories(categories) { const total = categories.reduce((s, c) => s + c.value, 0) || 1; const wrapper = document.getElementById('category-bars'); wrapper.innerHTML = categories.map(c => `

`).join(''); }
